# Código do arquivo CSV de PS4

games <- read.csv("C:/Users/24011626/Downloads/archive (1)/PS4_GamesSales.csv", stringsAsFactors = FALSE)

library(dplyr)

games_filtered <- games %>%
  filter(Genre %in% c("Action", "Sports"))
sales_by_year_genre <- games_filtered %>%
  group_by(Year, Genre) %>%
  summarise(Global = sum(Global, na.rm = TRUE)) %>%
  arrange(Year)

library(ggplot2)

ggplot(sales_by_year_genre, aes(x = Year, y = Global, color = Genre)) +
  geom_line(size = 1.2) +
  labs(
    title = "Evolução das Vendas Globais de Jogos (Ação vs Esporte)",
    x = "Ano de Lançamento",
    y = "Vendas Globais (em milhões)",
    color = "Gênero"
  ) +
  theme_minimal()
